# 02. Add Two Numbers

/**
 * Definition for singly-linked list.
 * public class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode(int x) { val = x; }
 * }
 */
class Solution {
    public ListNode addTwoNumbers(ListNode l1, ListNode l2) {
        
//         int[] lst = new int[]{2,4,3};
//         int i = 0;
    
//         ListNode r = l1;
//         while(l1 != null){
//             l1.val = lst[i];
//             l1 = l1.next;
//             i++;
//         }
//         l1 = r;
        
        ListNode sum = new ListNode(0);
        ListNode dummy = sum;
        int carry = 0;
        
        while(l1 != null || l2 != null){
            int x = 0,y = 0;
            if(l1 != null) x = l1.val;
            if(l2 != null) y = l2.val;
            int tmp = x + y + carry;
            if(tmp > 9){
                carry = tmp/10;// get carry
                tmp = tmp%10;// get tail
            }else{
                carry = 0;
            }
            sum.val = tmp;
            if(l1 != null) l1 = l1.next;
            if(l2 != null) l2 = l2.next;
            if(l1 != null || l2!=null){
                sum.next = new ListNode(0);
                sum = sum.next;
            }
        }
        if(carry != 0){
            sum.next = new ListNode(carry);  
            sum = sum.next;
        }
        return dummy;
    }
}
