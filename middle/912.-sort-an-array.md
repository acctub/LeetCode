Merge Sort in Regression

```java
//マージソート
class Solution {
    public int[] sortArray(int[] nums) {
        if(nums.length <= 1) return nums;
        int length = nums.length;
        int mid = length/2;
        //System.out.println("Length!"+length);
        
        int[] left = new int[mid];
        int[] right = new int[length-mid];
        
        for (int i = 0; i < length; i++){
            if (i < mid){
                left[i] = nums[i];
            }else{
                right[i-mid] = nums[i];
            }
        }
        //show(left,"left");
        //show(right,"right");
        left = sortArray(left);
        right = sortArray(right);
        int[] temp = merge(left, right);
        //show(temp,"temp");
        
        return temp;
        
    }
    public int[] merge(int[] left, int[] right){
        int i = 0, j = 0, p = 0;
        int [] temp = new int[left.length+right.length];
        
        for( ; i < left.length && j < right.length; p++){
            if(left[i] < right[j]){
                temp[p] = left[i];
                i++;
            }else{
                temp[p] = right[j];
                j++;
            }
        }
    
        for (; i < left.length; i++){
            temp[p++] = left[i];
        }
        for (; j < right.length; j++){
            temp[p++] = right[j];
        } 
        
        return temp;
    }
    public void show(int[] lst, String str){
        System.out.println(str+">");
        for(int i = 0; i < lst.length; i++){
            System.out.print(lst[i]);
        }
        System.out.println("<--");
    }
}
```

Quick Sort \(Clear\) O\(n\*logn\)

```java
class Solution {
    //public：クラス外部からもアクセスできる
    public int[] sortArray(int[] nums) {
        if(nums.length == 0)
            return nums;
        QuickSort(nums, 0, nums.length-1);
        return nums;
    }
    private void QuickSort(int[] nums, int L, int R){
        if(L >= R) return;
        //指標を選定し、指標より小さいものは左、大きいものは右
        int pivot = compare(nums, L, R);
        QuickSort(nums, L, pivot);
        QuickSort(nums, pivot+1, R);
    }
    private int compare(int[] nums, int L, int R){
        
        int pivot = nums[L];
        while(L < R){
            //指標より小さいものを探し、位置交換する
            while(R > L && nums[R] >= pivot) R--;
            nums[L] = nums[R];
            //指標より大きいものを探し、位置交換する
            while(L < R && nums[L] <= pivot) L++;
            nums[R] = nums[L];
        }
        //指標の次の位置は確定できた
        nums[L] = pivot;
        return L;
    }
}
```

BubbleSort O\(N^2\)

```java
private void bubbleSort(int[] nums){
        for(int i = nums.length-1; i >= 0; i--){
            for(int j = 0; j < i; j++){
                if(nums[j] > nums[j+1])
                    swap(nums, j, j+1);
            }
        }
    }
    private void swap(int[] nums, int i, int j){
        nums[i] = nums[i]^nums[j];//t = a^b = a xor b
        nums[j] = nums[j]^nums[i];
        nums[i] = nums[j]^nums[i];
    }
```

```python
class Solution(object):
    def sortArray(self, nums):
        if(nums == None or len(nums)==0):
            return nums
        self.QuickSort(nums, 0, len(nums)-1)
        return nums
        
    def QuickSort(self, nums, left, right):
        
        if(left >= right):
            return
        l,r = left, right
        pivot = nums[l]
        while(l < r):
            while(l < r and nums[r] >= pivot):
                r -= 1
            nums[l] = nums[r]
            while(l < r and nums[l] <= pivot):
                l += 1
            nums[r] = nums[l]
        nums[l] = pivot
        
        self.QuickSort(nums, left, l)
        self.QuickSort(nums, l+1, right)     
```

```python
def QuickSort(nums):
    if len(nums) <= 1:
        return nums
    
    pivot = nums[0]
    L = [i for i in nums if i < pivot]
    M = [i for i in nums if i == pivot]
    R = [i for i in nums if i > pivot]
    
    return QuickSort(L) + M + QuickSort(R)
```
